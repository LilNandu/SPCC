#include<stdio.h>
#include<ctype.h>
#include<string.h>

int n;
char prod[10][10];
char first[10][10];

// Add element to set (avoid duplicates)
void add(char *set, char val) {
    if(strchr(set, val) == NULL) {
        int len = strlen(set);
        set[len] = val;
        set[len+1] = '\0';
    }
}

// FIRST function
void findFirst(char c, int index) {
    if(!isupper(c)) {
        add(first[index], c);
        return;
    }

    for(int i=0; i<n; i++) {
        if(prod[i][0] == c) {
            if(prod[i][2] == '#') {
                add(first[index], '#');
            } else {
                findFirst(prod[i][2], index);
            }
        }
    }
}

int main() {
    printf("Enter number of productions: ");
    scanf("%d", &n);

    printf("Enter productions (Example: S=AB):\n");
    for(int i=0; i<n; i++) {
        scanf("%s", prod[i]);
    }

    // Initialize FIRST sets
    for(int i=0; i<10; i++) {
        first[i][0] = '\0';
    }

    // FIRST calculation
    for(int i=0; i<n; i++) {
        findFirst(prod[i][0], prod[i][0]-'A');
    }

    // Display FIRST sets
    printf("\nFIRST Sets:\n");
    for(int i=0; i<n; i++) {
        char c = prod[i][0];
        printf("FIRST(%c) = { ", c);

        for(int j=0; first[c-'A'][j] != '\0'; j++)
            printf("%c ", first[c-'A'][j]);

        printf("}\n");
    }

    return 0;
}
